def simple_sort(A):
    AA = []
    for j in range(len(A)-1):
        print(j)
        print(A)
        AA.extend(A)
        MIN, k = A[len(A)-j-1], -1
        for i in range(len(A)-j-2, -1, -1):
            if MIN > A[i]: 
                MIN = A[i]
                k = i
        if k > -1:
            A[k] = A[len(A)-j-1]
            A[len(A)-j-1] = MIN
    return AA

def bubble_sort(A):
    AA = []
    for j in range(len(A)-1):
        print(A)
        AA.extend(A)
        c = 0
        for i in range(len(A)-j-1):
            if A[i] < A[i+1]: 
                MIN = A[i]
                A[i] = A[i+1]
                A[i+1] = MIN
                c += 1
        if c < 1:  break
    return AA
