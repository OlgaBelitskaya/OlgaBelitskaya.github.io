@interact
def _(N=[2,3,4]):
    bd=digraphs.ButterflyGraph(N)
    nvl=[u+'$\\to$'+str(v) 
         for u,v in bd.vertices()]
    bd.relabel(dict(zip(bd.vertices(),nvl)))
    [bd.set_edge_label(u,v,v) 
     for u,v,l in bd.edges()]
    bd.show(figsize=7,vertex_size=3,
            layout='circular',
            color_by_label=True)
