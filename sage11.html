@interact
def _(n=[9,8,..,3],cm=['cool','jet','hsv']):
    var('r,phi,theta')
    def c(i,cmap): return colormaps[cmap](i/(1.*n))[:3]
    SE=SphericalElevation('elevation',['radius','azimuth'])
    p=sum([plot3d(pi/i,(r,0,1),(theta,0,2*pi),opacity=.1*i,
                  transformation=SE,color=c(i,cm)) 
           for i in [3..n]])
    p.show(frame=False)
