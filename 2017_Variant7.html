import pandas

def find_years(number, array):
    df = pandas.DataFrame(index=range(1980,2018,1), 
                          columns=['max_inc_tonnage_per_vagon', 
                                   'min_out_tonnage_per_vagon'])
    df= df.fillna(0).astype('float32')
    
    for i in range(number):
        
        year=int(array[i].split()[0].split('.')[1])
        tonnage_per_vagon=int(array[i].split()[1])/int(array[i].split()[2])
        
        if tonnage_per_vagon < 0:
            if abs(tonnage_per_vagon) < df['min_out_tonnage_per_vagon'][year]\
            or df['min_out_tonnage_per_vagon'][year] == 0:
                df['min_out_tonnage_per_vagon'][year] = abs(tonnage_per_vagon)
 
        else:
            if abs(tonnage_per_vagon) > df['max_inc_tonnage_per_vagon'][year]:
                df['max_inc_tonnage_per_vagon'][year] = abs(tonnage_per_vagon)
            
    years = []
    for year in range(1980,2018,1):
        if df['max_inc_tonnage_per_vagon'][year] > \
        df['min_out_tonnage_per_vagon'][year]:
            years.append(year)
    return years
