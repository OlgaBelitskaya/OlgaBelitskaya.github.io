if sum(numpy.sum(A == numpy.sort(A, axis=0), axis=0) < A.shape[0]) != 0:
    N = numpy.nonzero(numpy.sum(A == numpy.sort(A, axis=0), axis=0) < A.shape[0])[0][-1] + 1

    print(90*'=' + '\n' + '    N ( ): \n')
    print(N)
else:
    N = 0
    print(90*'=' + '\n' + '     )')
    
with open(fout, 'w') as f:
    for i in range(len(A)):
        f.write("{}\n".format(A[i]))
    f.write("{}\n".format(N))
