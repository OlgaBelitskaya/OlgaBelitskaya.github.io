import networkx as nx

elarge=[(u,v) for (u,v,d) in G.edges(data=True) if d['weight'] >6]
esmall=[(u,v) for (u,v,d) in G.edges(data=True) if d['weight'] <=6]                 

# positions for all nodes
pos=nx.shell_layout(G) 

# nodes
nx.draw_networkx_nodes(G,pos,node_size=700, node_color='steelblue')

# edges

                       width=6,alpha=0.5,style='dashed')

edge_labels=\
dict([((u,v,),d['weight']) for u,v,d in G.edges(data=True)])
