2. Уточненная постановка задачи.¶

B, n n .

$X_1, X_2, ..., X_n$ :

$\large{\displaystyle{X_i = \begin{cases} 1, \ B_{ii} > B_{ij}, \ \forall j: 0<j<n+1 \land j \neq i\\ 0, \ B_{ii} \leq B_{ij}, \ \forall j: 0<j<n+1 \land j \neq i\\ \end{cases}, \ \forall i: 0<i<n+1 }}$

$\large{\displaystyle{S = \sum_{\substack{0<i<n+1 \\ 0<j<n+1}} B_{ij} }}$

print(20*' ' + '.8\n' + 90*'=')

fin = str(input('     B: '))
fout = str(input('     B,  X    S: '))

print(90*'=' + '\n' + '  B: \n')
B = numpy.loadtxt(fin, dtype='int')
print(B)

X = numpy.array(len(B) * [0])
for i in range(len(B)):
    c = numpy.argmax(B[i])
    X[i] = (c == i)
print(90*'=' + '\n' + '  X: \n\n' + str(X))

S = numpy.sum(B)
print('\n' + '   B: \n\n' + str(S))

with open(fout, 'w') as f:
    for i in range(len(B)):
        f.write("{}\n".format(B[i]))
    f.write("{}\n".format(X))
    f.write("{}\n".format(S))
