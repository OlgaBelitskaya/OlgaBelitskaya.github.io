def _(N=(47,0,-1)):

    p+=sum([plot((sin(x),cos(x)),(pi*n/12,pi*(n+1)/12),
                 thickness=12,alpha=.7,
                 color=colormaps.jet(6*n)[:3]) 
            for n in [0..N]])
