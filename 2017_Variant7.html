Between the settlements A, B, C, D, E, F roads are constructed, the lengths of which are given in the table. Determine the length of the shortest path between points A and F (assuming that you can only move along the constructed roads).

import matplotlib.pyplot as plt
import networkx as nx

plt.figure(figsize=(18,10))
G=nx.Graph()

G.add_edge('A','B',weight=3)
G.add_edge('A','C',weight=8)
G.add_edge('A','E',weight=4)
G.add_edge('B','C',weight=5)
G.add_edge('B','D',weight=9)
G.add_edge('B','E',weight=2)
G.add_edge('B','F',weight=13)
G.add_edge('C','D',weight=4)
G.add_edge('C','E',weight=3)
G.add_edge('D','F',weight=3)
G.add_edge('E','F',weight=11)

elarge=[(u,v) for (u,v,d) in G.edges(data=True) if d['weight'] >6]
esmall=[(u,v) for (u,v,d) in G.edges(data=True) if d['weight'] <=6]                 

# positions for all nodes
pos=nx.shell_layout(G) 

# nodes
nx.draw_networkx_nodes(G,pos,node_size=700, node_color='steelblue')

# edges
nx.draw_networkx_edges(G,pos,edgelist=elarge,edge_color='darkred',
                       width=6,alpha=0.5,style='dashed')
nx.draw_networkx_edges(G,pos,edgelist=esmall,edge_color='darkblue',
                       width=3,alpha=0.5,style='dashed')
# labels
nx.draw_networkx_labels(G,pos,font_size=20,font_weight='bold')
edge_labels=\
dict([((u,v,),d['weight']) for u,v,d in G.edges(data=True)])
nx.draw_networkx_edge_labels(G,pos,font_size=20,font_weight='bold', 
                             label_pos=0.2, color='r', edge_labels=edge_labels)
