def read_txt(fin):
    with open(fin, 'r') as f:
        Y = f.readlines()
    Y = [x[1:-2].split() for x in Y]
    Y = [[int(x) for x in Y[i]] for i in range(len(Y))]
    return Matrix(Y[:int(len(Y)/2)]), Matrix(Y[int(len(Y)/2):])

read_in_txt <- function(fin) {
  Y <- read.csv(fin, header=F, sep=' ')
  Y[,1] <- as.integer(gsub("\\[", "", Y[,1]))
  Y[,length(Y)] <- as.integer(gsub("\\]", "", Y[,length(Y)]))
  A <- Y[1:length(Y),]
  B <- Y[(length(Y)+1):(2*length(Y)),]
  return(list(A, B))
}
