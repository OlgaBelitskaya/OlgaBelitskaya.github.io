find_logic_exp <- function(f) {
    c <- 0
    
    if (f(1,1,1) == 1 & f(0,1,1) == 1) {
        c <- c + 1
    }
    if (f(0,1,0) == 0 & f(0,0,0) == 0) {
        c <- c + 1
    }
    if (f(1,1,0) == 1 & f(1,1,1) == 1) {
        c <- c + 1
    }

    if (c == 3) {
        return(TRUE)
    } else {
        return(FALSE)
    }
}
