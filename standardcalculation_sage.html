def read_txt(fin):
    with open(fin, 'r') as f:
        Y = f.readlines()
    Y = [x[1:-2].split() for x in Y]
    Y = [[int(x) for x in Y[i]] for i in range(len(Y))]

    return Matrix(Y[:int(len(Y)/2)]), Matrix(Y[int(len(Y)/2):])
