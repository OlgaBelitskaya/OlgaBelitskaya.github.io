def create_sphere(center_x,center_y,center_z,radius):
    u, v = numpy.mgrid[0:2*numpy.pi:50j, 0:numpy.pi:50j]    
    x_grid = center_x + radius*numpy.cos(u)*numpy.sin(v)
    y_grid = center_x + radius*numpy.sin(u)*numpy.sin(v)
    z_grid = center_x + radius*numpy.cos(v)
    return x_grid,y_grid,z_grid

Xc,Yc,Zc = create_sphere(2.5,2.5,2.5,2.5)

def create_cylinder(center_x,center_y,radius,height):
    z = numpy.linspace(0, height, 50)
    theta = numpy.linspace(0, 2*numpy.pi, 50)
    theta_grid, z_grid = numpy.meshgrid(theta, z)
    x_grid = radius * numpy.cos(theta_grid) + center_x
    y_grid = radius * numpy.sin(theta_grid) + center_y
    return x_grid,y_grid,z_grid

Xc,Yc,Zc = create_cylinder(2.5,2.5,2.5,5)
ax.plot_surface(Xc, Yc, Zc, alpha=0.3, color='steelblue')

$R$. $2 * R$.

$S_{cylinder} = 2 * \pi * R^2 + 2 * \pi * R * 2 * R = 6 * \pi * R^2$

$S_{ball} = 4 * \pi * R^2 \implies S_{ball} \ / \ S_{cylinder} = 4 * \pi * R^2 \ / \ 6 * \pi * R^2 = 2 / 3$
