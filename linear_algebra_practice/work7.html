library(pracma)
problem_decision<-function(x){
    c(x[1]+x[4]-3, # A + B = C
      x[2]+x[5]-4, 
      x[3]+x[6]-1, 
      x[1]*1+x[2]*2+x[3]*1, # A ⊥ D 
      x[7]*1-x[4], # B || D 
      x[7]*2-x[5],
      x[7]*1-x[6])}
print('a1 a2 a3 b1 b2 b2 k')
round(fsolve(problem_decision,c(0,0,0,0,0,0,0))$x)
