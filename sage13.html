@interact
def _(N=[12,11,..,3]):
    def P(i): return [(cos(t),sin(t)) 
                      for t in srange(0,2*pi,2*pi/i)]
    def col(i): return colormaps.hsv(22*(i-3))[:3]
    p=sum([polygon2d(P(n),alpha=n/12,
                     color=col(n)).plot3d(z=-.15*n) 
           for n in [3..N]])
    p.show(frame=False)
