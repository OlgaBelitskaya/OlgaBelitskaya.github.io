program Lab5; //  lab5.dpr (DPR  Delphi PRoject)  
{$APPTYPE CONSOLE}
Uses Windows; //   (   -!)

    var
        i: integer;
        PRE, MIN: real;
        fin, fout: string;
        tfin, tfout: TextFile;
        A: array of real; 

    begin
        setConsoleCP(1251); //      (    -!)
        setConsoleOutputCP(1251); //   (    -!)
        
        if (ParamCount<2) then 
        begin 
        writeln(CritError:    );
        readln: exit;
        end; 

        {A0.1, A0.2, A0.3 - -  }
        writeln('.5' :40); {  c      .3}       
        
        {-  a A  .2.1, 2.2, 5.1, 5.2}
        writeln(' A: '); {      }
        
        AssignFile(tfin, ParamStr(1)); reset(tfin); {   } 
        AssignFile(tfout, ParamStr(2)); rewrite(tfout); {   }
                
        i := 1;             
        MIN := 10000000000;
                
        while not eof(tfin) do 

        begin
            SetLength(A,i);
            readln(tfin, A[i]); {      } 
            writeln(A[i]:5:6); {        }
            writeln(tfout, A[i]:5:6); {        }
            
            if i > 1 then
            begin
                if (A[i] + PRE < MIN) then MIN := A[i] + PRE;
                PRE: = A[i];
            end;
            
            i := i + 1;
        end;
            
        CloseFile(tfin); {   } 
    
        for i:=1 to 80 do 
            write('='); {        }
        writeln; 
        
        writeln('     A:', MIN:5:6);
        {          .6}             
        writeln(tfout, MIN:5:6); 
        { MIN         .6}
        
        CloseFile(tfout); {   }
        
    end.
