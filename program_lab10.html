    k = 0
    for i in range(len(A)):
        if A[i] % 9 == 0: 
            k = i + 1
    ci = []       
    for i in range(k, len(A)):
        if (A[i] < A[0]): 
            ci.append(i)    
    return ci, k

ci, k = conditions(A)

def max_condition(A, ci):
    j, MAX = 0, -1000000
    for i in range(len(A)):
        if i in ci:
            if (A[i] > MAX):
                MAX, j = A[i], i + 1
    return j

j = max_condition(A, ci)
