@interact
def _(N=[20,40,60],CM=['hsv','prism','jet']):
    def f(i): return (ln(i)*cos(i),ln(i)*sin(i))
    def c(i,cmap): return colormaps[cmap](i/(1.*N))[:3]
    p=sum([polygon([f(i),(0,f(i)[1]),(f(i)[0],0)],
                   color=c(i,CM),
                   alpha=.5).plot3d(z=-.1*i) 
           for i in [1..N]])
    p.show(frame=False)
